async function updateTimer(){ const { data } = await supabaseClient.rpc("get_active_round_info") if(!data || data.length === 0){ document.getElementById("countdown").innerText="Tournament finished" return } const secs = Math.max(0, data[0].remaining_seconds) const round = data[0].round const m = Math.floor(secs/60) const s = secs%60 const mm = String(m).padStart(2,"0") const ss = String(s).padStart(2,"0") document.getElementById("countdown").innerText=`${mm}:${ss}` // 👇 CLAVE: si llega a 0, forzamos avance if(secs===0){ await supabaseClient.rpc("advance_tournament") } }
